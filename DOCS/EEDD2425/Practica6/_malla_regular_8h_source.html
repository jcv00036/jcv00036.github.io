190typename MallaRegular<T>::Casilla *MallaRegular<T>::obtenerCasilla(float lon, float lat) {
191 if(lon > xMax || lon < xMin) throw std::invalid_argument("[MallaRegular::obtenerCasilla] Posición x fuera de los límites de la malla");
192 if(lat > yMax || lat < yMin) throw std::invalid_argument("[MallaRegular::obtenerCasilla] Posición y fuera de los límites de la malla");
193 int i = (lon - xMin) / tamaCasillaX;
194 int j = (lat - yMin) / tamaCasillaY;
195 return &mr[i][j];

199template<typename T>
200unsigned int MallaRegular<T>::distanciaPuntos(UTM &punto1, UTM &punto2) {
201 int R = 6378; //< Radio de la tierra en km
202 float lat1 = punto1.getLat(), lon1 = punto1.getLon(); //< Latitud y longitud del punto 1
203 float lat2 = punto2.getLat(), lon2 = punto2.getLon(); //< Latitud y longitud del punto 2
204
205 float deltaLat = (lat2 - lat1) * (M_PI / 180);
206 float deltaLon = (lon2 - lon1) * (M_PI / 180);
207
208 float a = std::pow(std::sin(deltaLat / 2), 2) + std::cos(lat1 * (M_PI / 180)) * std::cos(lat2 * (M_PI / 180)) * std::pow(std::sin(deltaLon / 2), 2);
209
210 float c = 2 * std::atan2(std::sqrt(a), std::sqrt(1 - a));
211
212 return R * c;
213}

271std::vector<T> MallaRegular<T>::buscarRadio(float xcentro, float ycentro, float radio) {
272 if(xcentro > xMax || xcentro < xMin)
273 throw std::invalid_argument("[MallaRegular::buscarRadio] Posición x fuera de los límites de la malla");
274 if(ycentro > yMax || ycentro < yMin)
275 throw std::invalid_argument("[MallaRegular::buscarRadio] Posición y fuera de los límites de la malla");
276
277 std::vector<T> resultado;
278
279 float radioCoords = radio / 111.1;
280 float limiteIzq = xcentro - radioCoords, limiteDer = xcentro + radioCoords;
281 float limiteInf = ycentro - radioCoords, limiteSup = ycentro + radioCoords;
282
283 if(limiteIzq < xMin) limiteIzq = xMin;
284 if(limiteDer > xMax) limiteDer = xMax;
285 if(limiteInf < yMin) limiteInf = yMin;
286 if(limiteSup > yMax) limiteSup = yMax;
287
288 for (float posX = limiteIzq; posX <= limiteDer; posX += tamaCasillaX) {
289 for (float posY = limiteInf; posY <= limiteSup; posY += tamaCasillaY) {
290 if (posX >= xMin && posX <= xMax && posY >= yMin && posY <= yMax) {
291 Casilla* casillaActual = obtenerCasilla(posX, posY);
292 if (!casillaActual) continue; // Esto lo que hace es ignorar las casillas nulas
293
294 for (const auto& elemento : casillaActual->puntos) {
295 UTM puntoCentro;
296 UTM puntoElemento;
297
298 puntoCentro.setLon(xcentro);
299 puntoCentro.setLat(ycentro);
300
301 puntoElemento.setLon(elemento->GetX());
302 puntoElemento.setLat(elemento->GetY());
303
304 float distanciaCalculada = distanciaPuntos(puntoCentro, puntoElemento);
305
306 if (distanciaCalculada <= radio) {
307 resultado.push_back(elemento);
308 }
309 }
310 }
311 }
312 }
313
314 return resultado;
